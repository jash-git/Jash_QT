這個是文字的跑馬燈喔，可以用在文件中喔！下面每種顏色的區域代表一種跑馬燈，你可以每種都試試看喔。
<SCRIPT Language="JavaScript">

var msg="這是用JAVA Script 所做出來的跑馬燈，高興怎麼用就怎麼用！ ";
var interval = 100;
var space10="";
var seq=0;

function Scroll() {
document.tmForm.tmText.value = msg.substring(seq, msg.length) + space10 + msg.substring(0, msg.length);
seq++;seq++;
if ( seq > msg.length ) { seq = 0 };
window.setTimeout("Scroll();", interval );
}

</SCRIPT>

var interval = 100;
var space10=" ";
var seq=0;

function Scroll1() {
document.tmForm.tmText.value = msg.substring(0, seq+1);
seq++;seq++;
if ( seq >= msg.length ) { seq = 0;
window.setTimeout("Scroll2();", interval );
}
else
window.setTimeout("Scroll1();", interval );
}
function Scroll2() {
document.tmForm.tmText.value = msg.substring(seq, msg.length) + space10 + msg.substring(0, msg.length);
seq++;seq++;
if ( seq > msg.length ) { seq = 0 };
window.setTimeout("Scroll2();", interval );

使用時只要將上面的程式碼複製，然貼在網頁中空白的地方(通常是貼在</HEAD>下面) 然後在<BODY>後面加上 OnLoad="Scroll();" 就可以啦！如下，
<BODY OnLoad="Scroll();" >

當然啦，後面可以繼續加上背景、顏色...等等標籤，像這樣。
<BODY OnLoad="Scroll();" BGCOLOR="#FFFF00 BACKGROUND="BG.GIF" LINK="#FFFFDD" ALINK="RED" VLINK="PURPLE">

低頭往下看看狀態列，是不是有一堆字跑出來啊，那就是狀態列走馬的應用喔。版豬收集了幾個狀態列走馬的JAVASCRIPT，讓網頁應用更靈活喔。

下面就是幾個走馬燈的形式，你可以選自己喜歡的。
<SCRIPT Language="JavaScript">
var msg="你看見我了嗎？我就是狀態列走馬燈的示範喔，打字加移動效果的喔！";
var interval = 100;
var spacelen = 120;
var space10=" ";
var seq=0;

function Scroll() {
len = msg.length;
window.status = msg.substring(0, seq+1);
seq++;
if ( seq >= len ) {
seq = spacelen;
window.setTimeout("Scroll2();", interval );
}
else
window.setTimeout("Scroll2();", interval );
}

function Scroll2() {
var out="";
for (i=1; i<=spacelen/space10.length; i++) out += space10;
out = out + msg;
len=out.length;
window.status=out.substring(seq, len);
seq++;
if ( seq >= len ) { seq = 0; };
window.setTimeout("Scroll2();", interval );
}
</SCRIPT>

使用時只要將上面的程式碼複製，然貼在網頁中空白的地方(通常是貼在</HEAD>下面) 然後在<BODY>後面加上 OnLoad="Scroll();" 就可以啦！如下，
<BODY OnLoad="Scroll();" >

當然啦，後面可以繼續加上背景、顏色...等等標籤，像這樣。
<BODY OnLoad="Scroll();" BGCOLOR="#FFFF00 BACKGROUND="BG.GIF" LINK="#FFFFDD" ALINK="RED" VLINK="PURPLE">
